# Plan 09-01: ML Dataset Builder & Feature Engineering

<objective>
Build the ML dataset preparation layer: a feature engineering service that reads `normalized_recommendations` from Supabase and converts each row into a numeric feature vector suitable for ML training. Export the feature schema as a JSON config file. Expose GET /api/ml/dataset endpoint to inspect the prepared dataset and GET /api/ml/features to view the feature schema.

**Design note:** XGBoost is Python-native. This project is a TypeScript/Node.js demo for a PUMA pitch. Rather than adding a Python sidecar dependency that complicates the demo environment, Phase 09 implements the full ML pipeline interface in TypeScript using a feature-weight approach that:
1. Is identical in API contract to a real XGBoost model
2. Uses the same feature vector structure that Phase 10 could swap for a real trained model
3. Produces realistic, differentiated success probabilities per shoe + foot profile combination
4. Can be upgraded to a real XGBoost model by replacing mlClassifierService.ts with a Python API call

This is documented transparently in the feature schema so future engineers know the upgrade path.
</objective>

<must_haves>
- Feature engineering function converts normalized_recommendations rows into numeric feature vectors
- Feature schema config defines all feature names, types, encoding (ordinal/one-hot), and ranges
- Dataset export returns feature vectors + labels (ml_success: 0/1) for all rows with feedback
- GET /api/ml/dataset returns { total_rows, labeled_rows, feature_schema, sample_rows }
- TypeScript compiles without errors
</must_haves>

/**
 * Convert a normalized_recommendations row into a numeric feature vector.
 */
export function buildFeatureVector(row: Record<string, unknown>): FeatureVector {
  const vec: FeatureVector = {
    // Width one-hot
    width_narrow: row['width'] === 'narrow' ? 1 : 0,
    width_standard: row['width'] === 'standard' ? 1 : 0,
    width_wide: row['width'] === 'wide' ? 1 : 0,
    // Arch one-hot
    arch_low: row['arch'] === 'low' ? 1 : 0,
    arch_medium: row['arch'] === 'medium' ? 1 : 0,
    arch_high: row['arch'] === 'high' ? 1 : 0,
    // Pronation one-hot
    pronation_neutral: row['pronation'] === 'neutral' ? 1 : 0,
    pronation_overpronation: row['pronation'] === 'overpronation' ? 1 : 0,
    pronation_supination: row['pronation'] === 'supination' ? 1 : 0,
    // Context (normalized)
    priority_score: Math.min(1, Math.max(0, ((row['priority_score'] as number) ?? 50) / 100)),
    hours_per_day: Math.min(1, Math.max(0, ((row['hours_per_day'] as number) ?? 4) / 16)),
    // Use case one-hot
    use_case_daily_commute: 0,
    use_case_office_wear: 0,
    use_case_running: 0,
    use_case_long_standing: 0,
    use_case_gym_casual: 0,
    use_case_rainy_conditions: 0,
    // Shoe model one-hot
    model_velocity_nitro_3: 0,
    model_foreverrun_nitro_2: 0,
    model_magnify_nitro_3: 0,
    model_deviate_nitro_3: 0,
    model_fast_r_2_elite: 0,
    model_electrify_nitro_3: 0,
    model_suede_xl: 0,
    model_rs_x: 0,
    model_rider_fv: 0,
    model_pwr_xx_nitro: 0,
    model_fuse_3: 0,
    // Rule engine score (normalized)
    primary_score: Math.min(1, Math.max(0, ((row['primary_score'] as number) ?? 0) / 100)),
  };

  // Set use case flag
  const useCaseKey = USE_CASE_KEY_MAP[row['use_case'] as string];

  // Set model flag
  const modelKey = MODEL_KEY_MAP[row['primary_model'] as string];
  if (modelKey) vec[modelKey] = 1;

  return vec;
}

// ─── Dataset Export ───────────────────────────────────────────────────────────

/**
 * Fetch all normalized_recommendations from Supabase and build the ML dataset.
 * Returns feature vectors + labels for all rows.
 * Rows without feedback have label=null (unlabeled — excluded from training).
 */
export async function buildMLDataset(): Promise<MLDataset> {
  const { data, error } = await supabase
    .from('normalized_recommendations')
    .select(
      'width, arch, pronation, use_case, priority_score, hours_per_day, primary_model, primary_score, ml_success'
    )
    .order('created_at', { ascending: false });

  if (error) {
    console.error('[MLDataset] Failed to fetch data:', error.message);
    return {
      total_rows: 0,
      labeled_rows: 0,
      unlabeled_rows: 0,
      feature_count: 32,
      class_balance: { success: 0, failure: 0 },
      feature_schema: getFeatureSchema(),
      sample_rows: [],
    };
  }

  const rows = data ?? [];
  let labeledRows = 0;
  let successCount = 0;
  let failureCount = 0;

  const allRows = rows.map((row: Record<string, unknown>) => {
    const features = buildFeatureVector(row);
    const label =
      row['ml_success'] === null || row['ml_success'] === undefined
        ? null
        : row['ml_success']
        ? 1
        : 0;

    if (label !== null) {
      labeledRows++;
      if (label === 1) successCount++;
      else failureCount++;
    }

    return { features, label };
  });

  return {
    total_rows: rows.length,
    labeled_rows: labeledRows,
    unlabeled_rows: rows.length - labeledRows,
    feature_count: getFeatureSchema().length,
    class_balance: { success: successCount, failure: failureCount },
    feature_schema: getFeatureSchema(),
    sample_rows: allRows.slice(0, 5), // first 5 as sample
  };
}

import { buildMLDataset, getFeatureSchema } from '../services/mlDatasetService';

const router = Router();

// GET /api/ml/dataset — fetch prepared ML dataset with feature vectors + labels
router.get('/dataset', async (_req: Request, res: Response) => {
  console.log('[ML Route] Building ML dataset...');
  const dataset = await buildMLDataset();
  res.json({ success: true, ...dataset });
});

// GET /api/ml/features — return the feature schema (column definitions)
router.get('/features', (_req: Request, res: Response) => {
  const schema = getFeatureSchema();
  res.json({
    success: true,
    feature_count: schema.length,
    features: schema,
    upgrade_path: 'Replace mlClassifierService.ts predict() with a Python XGBoost API call using these same feature vectors',
  });
});

export { router as mlRouter };
