# Plan 09-02: ML Classifier Service (Feature-Weight Scorer)

<objective>
Build the ML classifier service that takes a FeatureVector and returns a success probability (0-1). This implements ML-01 and ML-03 using a deterministic feature-weight scorer that:

1. Is designed with the same interface as a real trained XGBoost model
2. Uses hand-crafted weights derived from the shoe catalog attributes (wideFootScore, archSuitability, etc.) to produce realistic, differentiated success probabilities per shoe+profile combination
3. Returns a confidence score, model name, and feature importances for transparency
4. Has a clear UPGRADE_PATH comment showing how to replace it with a real XGBoost model

**Why not real XGBoost?** XGBoost is Python-native. Running it in Node.js requires either a Python microservice (adds infra complexity to a demo) or a JS port (less accurate). Since the demo has no real training data yet (feedback loops take weeks), a calibrated feature-weight scorer produces equivalent demo output. Phase 10 documents the full Python XGBoost integration path.
</objective>

<must_haves>
- `predictSuccessProbability(features: FeatureVector, shoeName: string): MLPrediction` function
- Returns: `{ probability: number, confidence: 'high'|'medium'|'low', model_version: string, top_features: string[] }`
- Probability is differentiated per shoe+foot profile (not a flat 0.8 for everything)
- Runs synchronously (no async needed — pure computation)
- TypeScript compiles without errors
</must_haves>

```typescript
import type { FeatureVector } from './mlDatasetService';
import { SHOES_CATALOG } from '../data/shoes';

// ─── ML Prediction Output ─────────────────────────────────────────────────────

export interface MLPrediction {
  probability: number;         // 0-1: predicted success probability
  confidence: 'high' | 'medium' | 'low';  // calibrated confidence band
  model_version: string;       // 'feature-weight-v1' | 'xgboost-v{N}' when upgraded
  top_features: string[];      // top 3 features driving this prediction
}

// ─── Shoe Attribute Lookup ─────────────────────────────────────────────────────

/**
 * Look up shoe catalog attributes for a given model name.
 * Used to derive biomechanical fit weights.
 */
function getShoeAttributes(modelName: string) {
  return SHOES_CATALOG.find(s => s.model === modelName)?.attributes ?? null;
}

// ─── Feature Importance Weights ───────────────────────────────────────────────

/**
 * Weights represent how much each feature group contributes to success probability.
 * Derived from domain knowledge about shoe biomechanics.
 *
 * UPGRADE_PATH: Replace this function with a call to a trained XGBoost model:
 *   const response = await fetch('http://ml-service/predict', { method: 'POST', body: JSON.stringify(features) });
 *   const { probability } = await response.json();
 * The FeatureVector schema in mlDatasetService.ts remains the same.
 */
const FEATURE_WEIGHTS = {
  // Foot-shoe fit is the most important factor
  footProfileMatch: 0.45,
  // Use case alignment
  useCaseAlignment: 0.30,
  // Rule engine score (proxy for overall match quality)
  ruleEngineScore: 0.25,
};

// ─── Predictor ────────────────────────────────────────────────────────────────

/**
 * Predict success probability for a shoe+foot profile combination.
 *
 * Algorithm:
 * 1. Compute foot-profile match score: does this shoe's attributes match the foot?
 *    - Width match: shoe.wideFootScore vs feature width flags
 *    - Arch match: shoe.archSuitability includes the foot arch?
 *    - Pronation match: shoe.pronationSupport includes foot pronation?
 * 2. Compute use-case alignment: does this shoe's use-case scores align with the context?
 * 3. Blend with the rule engine's primary_score (already captures most biomechanical match)
 * 4. Apply sigmoid-like scaling to keep output in 0.1–0.95 range (avoid overconfidence)
 */
export function predictSuccessProbability(
  features: FeatureVector,
  modelName: string
): MLPrediction {
  const attrs = getShoeAttributes(modelName);
  const topFeatures: string[] = [];

  // ── 1. Foot-profile match score (0-1) ──────────────────────────────────────

  let footScore = 0;

  if (attrs) {
    // Width compatibility
    let widthScore = 0;
    if (features.width_wide === 1) {
      widthScore = attrs.wideFootScore / 10;
      if (attrs.wideFootScore >= 7) topFeatures.push('wide_foot_compatibility');
    } else if (features.width_narrow === 1) {
      widthScore = (10 - attrs.wideFootScore) / 10;
      if (attrs.wideFootScore <= 3) topFeatures.push('narrow_foot_compatibility');
    } else {
      widthScore = 0.7; // standard width is usually fine
    }

    // Arch compatibility (binary: supported or not)
    let archScore = 0;
    const userArch =
      features.arch_low === 1 ? 'low' :
      features.arch_medium === 1 ? 'medium' : 'high';
    const archMatch = attrs.archSuitability.includes(userArch as 'low' | 'medium' | 'high');
    archScore = archMatch ? 0.9 : 0.3;
    if (archMatch) topFeatures.push('arch_support_match');

    // Pronation compatibility (binary: supported or not)
    let pronationScore = 0;
    const userPronation =
      features.pronation_neutral === 1 ? 'neutral' :
      features.pronation_overpronation === 1 ? 'overpronation' : 'supination';
    const pronationMatch = attrs.pronationSupport.includes(userPronation as 'neutral' | 'overpronation' | 'supination');
    pronationScore = pronationMatch ? 0.9 : 0.2;
    if (pronationMatch && userPronation !== 'neutral') topFeatures.push('pronation_support_match');

    footScore = (widthScore + archScore + pronationScore) / 3;
  } else {
    footScore = 0.5; // unknown shoe → neutral
  }

  // ── 2. Use-case alignment score (0-1) ─────────────────────────────────────

  let useCaseScore = 0;

  if (attrs) {
    // Map use case flags to relevant shoe attribute scores
    let relevantAttrScore = 5; // default mid

    if (features.use_case_running === 1) {
      relevantAttrScore = attrs.runningScore;
      if (attrs.runningScore >= 7) topFeatures.push('running_performance');
    } else if (features.use_case_daily_commute === 1 || features.use_case_office_wear === 1) {
      relevantAttrScore = attrs.dailyCommuteScore;
      if (attrs.dailyCommuteScore >= 7) topFeatures.push('commute_comfort');
    } else if (features.use_case_long_standing === 1) {
      relevantAttrScore = attrs.longWearComfortScore;
      if (attrs.longWearComfortScore >= 8) topFeatures.push('long_wear_comfort');
    } else if (features.use_case_rainy_conditions === 1) {
      relevantAttrScore = attrs.wetGripScore;
      if (attrs.wetGripScore >= 6) topFeatures.push('wet_grip_capability');
    } else if (features.use_case_gym_casual === 1) {
      relevantAttrScore = (attrs.runningScore + attrs.dailyCommuteScore) / 2;
    }

    // Long wear hours boost comfort score importance
    if (features.hours_per_day > 0.5 && attrs.longWearComfortScore >= 7) {
      relevantAttrScore = (relevantAttrScore + attrs.longWearComfortScore) / 2;
      topFeatures.push('high_wear_duration_comfort');
    }

    useCaseScore = Math.min(1, relevantAttrScore / 10);
  } else {
    useCaseScore = 0.5;
  }

  // ── 3. Rule engine score (already normalized 0-1) ────────────────────────

  const ruleScore = features.primary_score;
  if (ruleScore > 0.75) topFeatures.push('high_rule_engine_match');

  // ── 4. Weighted blend ─────────────────────────────────────────────────────

  const rawProbability =
    footScore * FEATURE_WEIGHTS.footProfileMatch +
    useCaseScore * FEATURE_WEIGHTS.useCaseAlignment +
    ruleScore * FEATURE_WEIGHTS.ruleEngineScore;

  // Scale to 0.1–0.95 range (avoid extreme confidence in a demo with no real training data)
  const probability = Math.round((0.1 + rawProbability * 0.85) * 100) / 100;

  // ── 5. Confidence band ────────────────────────────────────────────────────

  const confidence: 'high' | 'medium' | 'low' =
    probability >= 0.75 ? 'high' :
    probability >= 0.50 ? 'medium' : 'low';

  // Keep top 3 unique features
  const uniqueTopFeatures = [...new Set(topFeatures)].slice(0, 3);
  if (uniqueTopFeatures.length === 0) uniqueTopFeatures.push('overall_compatibility');

  return {
    probability,
    confidence,
    model_version: 'feature-weight-v1',
    top_features: uniqueTopFeatures,
  };
}

/**
 * Batch predict success probabilities for a list of shoe models.
 * Used by the hybrid scoring engine to get ML scores for all candidates.
 */
export function batchPredictSuccessProbabilities(
  features: FeatureVector,
  modelNames: string[]
): Map<string, MLPrediction> {
  const predictions = new Map<string, MLPrediction>();
  for (const modelName of modelNames) {
    predictions.set(modelName, predictSuccessProbability(features, modelName));
  }
  return predictions;
}
```
