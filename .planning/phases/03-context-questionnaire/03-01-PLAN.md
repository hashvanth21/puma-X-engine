## Tasks

<task id="03-01-T1">
<title>Create useQuestionnaire step engine hook</title>
<action>
Create `puma-frontend/src/hooks/useQuestionnaire.ts`.
This hook manages the current step index, handles forward/back navigation, and bridges to the Zustand `questionnaireStore`.

```typescript
import { useState, useCallback } from 'react';
import { useAppStore } from '@/stores';
import type { UserContext } from '@/types';

const TOTAL_STEPS = 5;

export function useQuestionnaire() {
  const [step, setStep] = useState(0);
  const setContextField = useAppStore((s) => s.setContextField);
  const context = useAppStore((s) => s.context);

  const next = useCallback(() => {
    setStep((s) => Math.min(s + 1, TOTAL_STEPS - 1));
  }, []);

  const back = useCallback(() => {
    setStep((s) => Math.max(s - 1, 0));
  }, []);

  const select = useCallback(
    <K extends keyof UserContext>(key: K, value: UserContext[K]) => {
      setContextField(key, value);
      // Auto-advance after 400ms
      setTimeout(() => next(), 400);
    },
    [setContextField, next],
  );

  const isFirst = step === 0;
  const isLast = step === TOTAL_STEPS - 1;

  return { step, totalSteps: TOTAL_STEPS, next, back, select, isFirst, isLast, context };
}
```

<title>Create StepProgress dot indicator</title>
<action>
Create `puma-frontend/src/screens/Screen3Questions/StepProgress.tsx`.
Render 5 circular dots (filled for completed steps, accent-colored for current, muted for upcoming). Use Framer Motion `motion.div` with scale animations for the active dot.

```typescript
import { motion } from 'framer-motion';

interface StepProgressProps {
  current: number;
  total: number;
}

export const StepProgress = ({ current, total }: StepProgressProps) => (
  <div className="flex items-center justify-center gap-2">
    {Array.from({ length: total }).map((_, i) => (
      <motion.div
        key={i}
        className={`rounded-full ${
          i < current
            ? 'bg-accent w-2 h-2'
            : i === current
              ? 'bg-accent w-3 h-3'
              : 'bg-bg-elevated w-2 h-2'
        }`}
        animate={{ scale: i === current ? 1.2 : 1 }}
        transition={{ type: 'spring', stiffness: 300, damping: 20 }}
      />
    ))}
    <span className="ml-2 text-text-muted text-xs">{current + 1}/{total}</span>
  </div>
);
```
